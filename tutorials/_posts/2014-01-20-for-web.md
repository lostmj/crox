## 加载Crox

```html
<!-- 加载 Crox -->
<script src="http://g.tbcdn.cn/thx/crox/1.2.0/crox-all.js"></script>
```

## 使用 Crox


```js
var tmpl = '{{root.a}} - {{root.b}}';

// 编译成原生js Function
var fn = Crox.compile(tmpl);

var html = fn({
    a: 1,
    b: 2
});

console.log(html);  // 1 - 2

```
