title: 通过set设置临时变量
---

{% raw %}

```js
var data = {
    b: 2,
    bb: {
        bbb: {
            bbbb: {
                bbbbb: 1
            }
        }
    }
};

var tmpl = "{{root.b}} \
{{set bbbb = root.bb.bbb.bbbb}}\
- {{bbbb.bbbbb}}\
- {{root.bb.bbb.bbbb.bbbbb}}"

console.log(Crox.render(tmpl, data)); // 2 - 1 - 1
```
