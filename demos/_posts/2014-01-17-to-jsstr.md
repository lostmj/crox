```
function anonymous(root) {

    var obj = { '<': '&lt;', '>': '&gt;', '&': '&amp;', '"': '&quot;' };

    function $htmlEncode(s) {
        return String(s).replace(/[<>&"]/g, function(c) {
            return obj[c];
        });
    }

    var $s = '';
    function $print(s){ $s += s; }    

    $print($htmlEncode(root.a));
    $print(" - ");
    $print($htmlEncode(root.b));

    return $s;
}
```
